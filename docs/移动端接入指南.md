## 一、技术选型

### 1.1 总体架构

**uni-app + 原生插件混合架构**：电商主链路用 uni-app 统一开发，3D/AR/支付/登录等强平台依赖功能通过原生插件实现。

| 层级 | 技术方案 | 占比 |
| :--- | :--- | :---: |
| **主框架** | uni-app（Vue 3 + TypeScript） | 80%+ |
| **Android 原生插件** | Kotlin + Jetpack Compose | ~10% |
| **iOS 原生插件** | Swift + SwiftUI | ~10% |
| **状态管理** | Pinia | - |
| **网络请求** | uni.request 封装（luch-request） | - |
| **编译目标** | iOS App / Android App / H5（PWA） | - |

### 1.2 模块技术方案

| 模块 | 技术方案 | 原因 |
| :--- | :--- | :--- |
| 首页 / 商品 / 分类 | uni-app | 标准电商 UI |
| 购物车 / 订单 / 会员 | uni-app | 表单 + 列表 |
| 社区（笔记 / 评论 / 直播） | uni-app | 信息流 + 视频播放 |
| 消息中心 / 客服 | uni-app | 列表 + 聊天 UI |
| **3D Pet Hub** | **原生插件** | 3D 渲染、手势、骨骼动画 |
| **AR 试穿** | **原生插件** | ARKit / ARCore |
| **3D 装扮商城** | **原生插件** | 3D 模型换装 |
| **支付 SDK** | **原生插件** | Stripe / PayPal / Apple Pay / Google Pay |
| **社交登录** | **原生插件** | Apple / Google / Facebook |
| **推送通知** | **原生插件** | FCM / APNs |
| **扫码 / 拍图搜索** | **原生插件** | ML Kit / Vision |
| Apple/Google Wallet | 原生插件 | PassKit / Google Wallet API |
| 复杂动画（Paw Tab） | uni-app Lottie 插件 | 简单 Lottie 动画 |

### 2.3 命名规范

**uni-app 层（Vue 3 + TypeScript）**
- 组件 / 页面：**PascalCase**（如 `HomePage.vue`、`ProductCard.vue`）
- 函数 / 变量：**camelCase**（如 `fetchGoodsList`、`productList`）
- 常量：**SNAKE_CASE**（如 `MAX_RETRY_COUNT`）
- Pinia Store：**camelCase** + Store 后缀（如 `useUserStore`）
- 工具类文件：**camelCase**（如 `request.ts`、`storage.ts`）
- 静态资源：`snake_case`（如 `ic_home.png`、`banner_default.jpg`）

**原生插件层**
- Android：类名 PascalCase，函数/变量 camelCase，常量 SNAKE_CASE
- iOS：类名 PascalCase，函数/变量 camelCase，常量 camelCase
- 缩进：**2 个空格**（不要 Tab）
- 注释：中文，关键逻辑必须有

### 3.3 网络请求封装

```typescript
// src/api/request.ts
// 基于 luch-request 封装的统一请求客户端
import Request, { HttpRequestConfig, HttpResponse } from 'luch-request'
import { useUserStore } from '@/stores/user'

// 创建请求实例
const http = new Request({
  baseURL: import.meta.env.VITE_API_BASE_URL + '/api/v1',
  timeout: 15000,
  header: {
    'Content-Type': 'application/json'
  }
})

// 请求拦截器
http.interceptors.request.use(
  (config: HttpRequestConfig) => {
    const userStore = useUserStore()
    // Token 注入
    if (userStore.accessToken) {
      config.header.Authorization = `Bearer ${userStore.accessToken}`
    }
    // 请求签名
    if (config.method === 'POST') {
      config.header['X-Sign'] = generateSign(config.data)
      config.header['X-Timestamp'] = Date.now().toString()
    }
    return config
  },
  (error: any) => Promise.reject(error)
)

// 响应拦截器
http.interceptors.response.use(
  (response: HttpResponse) => {
    const { code, data, message } = response.data
    if (code === 0) {
      return data
    } else if (code === 1001) {
      // Token 失效，跳转登录
      const userStore = useUserStore()
      userStore.logout()
      uni.navigateTo({ url: '/pages/login/LoginPage' })
      return Promise.reject(new Error('Token expired'))
    } else {
      uni.showToast({ title: message || '请求失败', icon: 'none' })
      return Promise.reject(new Error(message))
    }
  },
  (error: any) => {
    uni.showToast({ title: '网络异常，请稍后重试', icon: 'none' })
    return Promise.reject(error)
  }
)

// 生成请求签名
function generateSign(data: any): string {
  const params = Object.keys(data).sort().map(k => `${k}=${data[k]}`).join('&')
  const signStr = params + '&key=' + import.meta.env.VITE_SIGN_KEY
  return sha256(signStr)
}

export default http
```

### 3.4 Pinia 状态管理

```typescript
// src/stores/user.ts
// 用户状态管理
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import http from '@/api/request'

export const useUserStore = defineStore('user', () => {
  // 状态
  const accessToken = ref<string>('')
  const refreshToken = ref<string>('')
  const userProfile = ref<any>(null)
  const isLogin = computed(() => !!accessToken.value)

  // 登录
  async function login(params: LoginParams) {
    const res: any = await http.post('/user/login', params)
    accessToken.value = res.accessToken
    refreshToken.value = res.refreshToken
    // 持久化存储
    uni.setStorageSync('access_token', res.accessToken)
    uni.setStorageSync('refresh_token', res.refreshToken)
    await fetchProfile()
  }

  // 获取用户信息
  async function fetchProfile() {
    const res: any = await http.get('/user/profile')
    userProfile.value = res
  }

  // 退出登录
  function logout() {
    accessToken.value = ''
    refreshToken.value = ''
    userProfile.value = null
    uni.removeStorageSync('access_token')
    uni.removeStorageSync('refresh_token')
  }

  // Token 刷新
  async function refreshTokenFn() {
    try {
      const res: any = await http.post('/user/refresh-token', {
        refreshToken: refreshToken.value
      })
      accessToken.value = res.accessToken
      uni.setStorageSync('access_token', res.accessToken)
      return res.accessToken
    } catch {
      logout()
      throw new Error('Token refresh failed')
    }
  }

  return {
    accessToken,
    refreshToken,
    userProfile,
    isLogin,
    login,
    fetchProfile,
    logout,
    refreshTokenFn
  }
})
```

```typescript
// MOYUYO-Payment 支付插件封装
// 统一支付入口，隐藏原生平台差异

import http from '@/api/request'

// 检查插件是否可用
function isPaymentPluginAvailable(): boolean {
  // #ifdef APP-PLUS
  return !!uni.requireNativePlugin('MOYUYO-Payment')
  // #endif
  // #ifndef APP-PLUS
  return false
  // #endif
}

// 创建支付
export async function createPay(orderNo: string, payMethod: string): Promise<boolean> {
  // 先调用后端创建支付单
  const res: any = await http.post('/pay/create', { orderNo, payMethod })

  // #ifdef APP-PLUS
  const payment = uni.requireNativePlugin('MOYUYO-Payment')

  return new Promise((resolve, reject) => {
    switch (payMethod) {
      case 'STRIPE':
        payment.payWithStripe({
          clientSecret: res.clientSecret,
          customerId: res.customerId,
          ephemeralKey: res.ephemeralKey
        }, (result: any) => {
          if (result.success) resolve(true)
          else reject(new Error(result.message))
        })
        break
      case 'PAYPAL':
        payment.payWithPayPal({
          orderId: res.paypalOrderId
        }, (result: any) => {
          if (result.success) resolve(true)
          else reject(new Error(result.message))
        })
        break
      case 'APPLE_PAY':
        // #ifdef IOS
        payment.payWithApplePay({
          merchantId: res.merchantId,
          amount: res.amount,
          currency: 'USD'
        }, (result: any) => {
          if (result.success) resolve(true)
          else reject(new Error(result.message))
        })
        // #endif
        // #ifdef ANDROID
        reject(new Error('Apple Pay not supported on Android'))
        // #endif
        break
      case 'GOOGLE_PAY':
        // #ifdef ANDROID
        payment.payWithGooglePay({
          clientSecret: res.clientSecret
        }, (result: any) => {
          if (result.success) resolve(true)
          else reject(new Error(result.message))
        })
        // #endif
        // #ifdef IOS
        reject(new Error('Google Pay not supported on iOS'))
        // #endif
        break
      default:
        reject(new Error('Unsupported payment method'))
    }
  })
  // #endif

  // #ifndef APP-PLUS
  throw new Error('Payment plugin not available on this platform')
  // #endif
}
```

```typescript
// Apple 登录
export async function signInWithApple(): Promise<void> {
  // #ifdef APP-PLUS
  const auth = uni.requireNativePlugin('MOYUYO-Auth')

  return new Promise((resolve, reject) => {
    // #ifdef IOS
    auth.signInWithApple(async (result: any) => {
      if (result.success) {
        const userStore = useUserStore()
        const res: any = await http.post('/user/login/apple', {
          identityToken: result.identityToken,
          authorizationCode: result.authorizationCode
        })

      } else {
        reject(new Error(result.message))
      }
```

### 4.5 通用 ViewModel 模式

```kotlin
// BaseViewModel.kt
// 统一 UI 状态封装
sealed class UiState<out T> {
  data object Loading : UiState<Nothing>()
  data class Success<T>(val data: T) : UiState<T>()
  data class Error(val code: Int, val message: String) : UiState<Nothing>()
}

abstract class BaseViewModel : ViewModel() {

  /** 统一处理网络请求 */
  protected suspend fun <T> apiCall(
    showLoading: Boolean = true,
    block: suspend () -> ApiResponse<T>
  ): UiState<T> {
    return try {
      if (showLoading) _loading.value = true
      val resp = block()
      if (resp.code == 0 && resp.data != null) {
        UiState.Success(resp.data)
      } else {
        UiState.Error(resp.code, resp.message).also {
          handleBizError(resp.code, resp.message)
        }
      }
    } catch (e: Exception) {
      UiState.Error(-1, e.message ?: "网络异常").also {
        // 上报埋点
      }
    } finally {
      _loading.value = false
    }
  }

  private val _loading = MutableStateFlow(false)
  val loading: StateFlow<Boolean> = _loading

  /** 业务错误全局处理（Token 失效等） */
  private fun handleBizError(code: Int, message: String) {
    if (code == 1001) {
      // Token 失效，跳转登录
      Router.jump(RoutePath.LOGIN)
    }
  }
}
```

#### 4.9.3 Apple 登录（Android 端）

```kotlin
// AppleAuthHelper.kt
// Apple 登录（通过 Custom Tabs / WebView + PKCE OAuth 2.0 流程）
// Apple 无 Android 原生 SDK，使用网页 OAuth 流程
package com.moyuyo.app.auth

import android.app.Activity
import android.net.Uri
import android.util.Base64
import androidx.browser.customtabs.CustomTabsIntent
import java.security.MessageDigest
import java.security.SecureRandom
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class AppleAuthHelper @Inject constructor(
  private val api: ApiService
) {
  // Apple OAuth 配置
  companion object {
    private const val CLIENT_ID = "com.moyuyo.app.client"
    private const val REDIRECT_URI = "https://api.moyuyo.com/v1/auth/apple/callback"
    private const val AUTH_URL = "https://appleid.apple.com/auth/authorize"
    private const val SCOPE = "name email"
    private const val RESPONSE_TYPE = "code"
    private const val RESPONSE_MODE = "query"
  }

  private var state: String = ""
  private var codeVerifier: String = ""

  // 生成 PKCE code_verifier（43~128 字符随机串）
  private fun generateCodeVerifier(): String {
    val random = SecureRandom()
    val bytes = ByteArray(32)
    random.nextBytes(bytes)
    return Base64.encodeToString(
      bytes,
      Base64.URL_SAFE or Base64.NO_WRAP or Base64.NO_PADDING
    )
  }

  // 生成 SHA256 哈希的 code_challenge
  private fun generateCodeChallenge(verifier: String): String {
    val digest = MessageDigest.getInstance("SHA-256")
    val hash = digest.digest(verifier.toByteArray(Charsets.UTF_8))
    return Base64.encodeToString(
      hash,
      Base64.URL_SAFE or Base64.NO_WRAP or Base64.NO_PADDING
    )
  }

  // 生成防 CSRF state
  private fun generateState(): String {
    val random = SecureRandom()
    val bytes = ByteArray(16)
    random.nextBytes(bytes)
    return Base64.encodeToString(
      bytes,
      Base64.URL_SAFE or Base64.NO_WRAP or Base64.NO_PADDING
    )
  }

  // 启动 Apple 登录（使用 Custom Tabs 打开授权页）
  fun startSignIn(activity: Activity) {
    state = generateState()
    codeVerifier = generateCodeVerifier()
    val codeChallenge = generateCodeChallenge(codeVerifier)

    val uri = Uri.parse(AUTH_URL).buildUpon()
      .appendQueryParameter("client_id", CLIENT_ID)
      .appendQueryParameter("redirect_uri", REDIRECT_URI)
      .appendQueryParameter("response_type", RESPONSE_TYPE)
      .appendQueryParameter("response_mode", RESPONSE_MODE)
      .appendQueryParameter("scope", SCOPE)
      .appendQueryParameter("state", state)
      .appendQueryParameter("code_challenge", codeChallenge)
      .appendQueryParameter("code_challenge_method", "S256")
      .build()

    val customTabsIntent = CustomTabsIntent.Builder().build()
    customTabsIntent.launchUrl(activity, uri)
  }

  // 处理 Apple 回调（从 Redirect URI 中提取 code 和 state）
  suspend fun handleCallback(uri: Uri): Result<AppleAuthResult> {
    // 校验 state 防 CSRF
    val returnedState = uri.getQueryParameter("state")
    if (returnedState != state) {
      return Result.failure(IllegalStateException("State mismatch, possible CSRF attack"))
    }

    // 检查是否有错误
    val error = uri.getQueryParameter("error")
    if (error != null) {
      val errorDesc = uri.getQueryParameter("error_description") ?: error
      return Result.failure(Exception("Apple auth error: $errorDesc"))
    }

    // 获取授权码
    val code = uri.getQueryParameter("code")
      ?: return Result.failure(Exception("No authorization code returned"))

    // 将 code 和 code_verifier 发送到后端，由后端换取 token 并完成登录
    return try {
      val resp = api.appleLogin(
        code = code,
        codeVerifier = codeVerifier,
        redirectUri = REDIRECT_URI
      )
      Result.success(AppleAuthResult(token = resp.token, user = resp.user))
    } catch (e: Exception) {
      Result.failure(e)
    }
  }
}

data class AppleAuthResult(
  val token: String,
  val user: UserProfile
)
```

```kotlin
// QrCodeEncoder.kt
// otpauth URI -> Bitmap；Hirschmann 编码可避免 Base64 异常字符
object QrCodeEncoder {
  fun encodeHirschmann(contents: String, sizePx: Int): Bitmap {
    val matrix = MultiFormatWriter().encode(
      contents, BarcodeFormat.QR_CODE, sizePx, sizePx,
      mapOf(EncodeHintType.MARGIN to 1)
    )
    val w = matrix.width
    val pixels = IntArray(w * matrix.height) { i ->
      if (matrix.get(i % w, i / w)) 0xFF000000.toInt() else 0xFFFFFFFF.toInt()
    }
    return Bitmap.createBitmap(w, matrix.height, Bitmap.Config.ARGB_8888).apply {
      setPixels(pixels, 0, w, 0, 0, w, matrix.height)
    }
  }
}
```

```kotlin
// ConsentViewModel.kt
// 实现"防抖提交"：用户切换开关时不立即请求，离开页面或停止操作 1.5s 后再 PATCH
@HiltViewModel
class ConsentViewModel @Inject constructor(
  private val api: ApiService
) : ViewModel() {

  private val _uiState = MutableStateFlow<ConsentUiState>(ConsentUiState.Loading)
  val uiState: StateFlow<ConsentUiState> = _uiState

  private val pending = MutableStateFlow(ConsentPatch())
  private val submitChannel = Channel<Unit>(Channel.CONFLATED)

  init {
    reload()
    // 防抖：仅发送最后一次变更
    viewModelScope.launch {
      submitChannel.consumeAsFlow()
        .debounce(1500)
        .collect { submit() }
    }
  }

  fun reload() {
    viewModelScope.launch {
      _uiState.value = ConsentUiState.Loading
      when (val r = apiCall { api.getConsent() }) {
        is UiState.Success -> _uiState.value = ConsentUiState.Loaded(
          marketingEmail = r.data.marketingEmail,
          marketingPush = r.data.marketingPush,
          analyticsCookies = r.data.analyticsCookies,
          advertisingCookies = r.data.advertisingCookies,
          lastUpdatedAt = r.data.lastUpdatedAt
        )
        is UiState.Error -> _uiState.value = ConsentUiState.Error(r.message)
        else -> Unit
      }
    }
  }

  fun update(transform: (ConsentPatch) -> ConsentPatch) {
    pending.update(transform)
    submitChannel.trySend(Unit)
  }

  private suspend fun submit() {
    val snap = pending.value
    when (val r = apiCall { api.putConsent(snap) }) {
      is UiState.Success -> (_uiState.value as? ConsentUiState.Loaded)?.let {
        _uiState.value = it.copy(lastUpdatedAt = r.data.lastUpdatedAt)
      }
      else -> Unit
    }
  }
}

data class ConsentPatch(
  val marketingEmail: Boolean = false,
  val marketingPush: Boolean = false,
  val analyticsCookies: Boolean = false,
  val advertisingCookies: Boolean = false
)

  data object Loading : ConsentUiState

    val marketingPush: Boolean,
    val analyticsCookies: Boolean,
```

```swift
// APIClient.swift
// 网络请求客户端
final class APIClient {
  static let shared = APIClient()
  private let session: Session

  private init() {
    // 注入拦截器：公共头、Token、签名
    let interceptors = Interceptor(
      adapter: HeaderAdapter(),
      retrier: AuthRetrier()
    )
    session = Session(interceptor: interceptors)
  }

  func request<T: Decodable>(
    _ route: APIRouter,
    as type: T.Type
  ) async throws -> T {
    let url = Config.apiBaseURL + Config.apiVersion + route.path
    let response = try await session.request(
      url,
      method: route.method,
      parameters: route.parameters,
      encoding: route.method == .get ? URLEncoding.queryString : JSONEncoding.default
    )
    .validate()
    .serializingDecodable(ApiResponse<T>.self)
    .value
    guard response.code == 0, let data = response.data else {
      throw APIError.bizError(code: response.code, message: response.message)
    }
    return data
  }
}
```

```swift
import AuthenticationServices

@MainActor
final class AppleAuthHelper: NSObject, ASAuthorizationControllerDelegate {
  private let api: APIClient
  private var continuation: CheckedContinuation<ASAuthorization, Error>?

  init(api: APIClient) {
    self.api = api
  }

  /// 启动 Apple 登录流程
  func signIn() async throws -> ASAuthorization {
    let provider = ASAuthorizationAppleIDProvider()
    let request = provider.createRequest()
    request.requestedScopes = [.fullName, .email]

    return try await withCheckedThrowingContinuation { cont in
      self.continuation = cont
      let controller = ASAuthorizationController(authorizationRequests: [request])
      controller.delegate = self
      controller.presentationContextProvider = self
      controller.performRequests()
    }
  }

  /// 授权成功回调
  func authorizationController(
    controller: ASAuthorizationController,
    didCompleteWithAuthorization authorization: ASAuthorization
  ) {
    continuation?.resume(returning: authorization)
    continuation = nil
  }

  /// 授权失败回调
  func authorizationController(
    controller: ASAuthorizationController,
    didCompleteWithError error: Error
  ) {
    continuation?.resume(throwing: error)
    continuation = nil
  }
}

extension AppleAuthHelper: ASAuthorizationControllerPresentationContextProviding {
  func presentationAnchor(
    for controller: ASAuthorizationController
  ) -> ASPresentationAnchor {
    UIApplication.shared.connectedScenes
      .compactMap { $0 as? UIWindowScene }
      .flatMap { $0.windows }
      .first { $0.isKeyWindow } ?? UIWindow()
  }
}
```

```swift
final class EmailAuthViewModel: ObservableObject {
  @Published var email: String = ""
  @Published var password: String = ""
  @Published var isLoading: Bool = false
  @Published var errorMessage: String?
  @Published var requires2FA: Bool = false

  private let tokenStore: TokenStore

  init(api: APIClient, tokenStore: TokenStore) {
    self.api = api
    self.tokenStore = tokenStore
  }

  func signIn() async {
    isLoading = true
    defer { isLoading = false }
    do {
      let resp = try await api.loginWithEmail(email: email, password: password)
      if resp.require2fa, let token = resp.tempToken {
        self.tempToken = token
        self.requires2FA = true
      } else {
        tokenStore.save(access: resp.accessToken, refresh: resp.refreshToken)
        // 跳转首页
      }
    } catch {
      errorMessage = error.localizedDescription
    }
  }

  func verify2FA(code: String) async {
    isLoading = true
    defer { isLoading = false }
    do {
      guard let token = tempToken else { return }
      let resp = try await api.verify2FA(tempToken: token, code: code)
      tokenStore.save(access: resp.accessToken, refresh: resp.refreshToken)
      requires2FA = false
      // 跳转首页
    } catch {
      errorMessage = error.localizedDescription
    }
  }
```

```swift
// 用 CoreImage 生成 otpauth URI 的二维码；不依赖第三方
import CoreImage
import UIKit

enum QRCodeRenderer {
  static func render(string: String, size: CGFloat) -> UIImage? {
    let filter = CIFilter.qrCodeGenerator()
    filter.message = Data(string.utf8)
    filter.correctionLevel = "M"
    guard let output = filter.outputImage else { return nil }
    let scaleX = size / output.extent.size.width
    let scaleY = size / output.extent.size.height
    let transformed = output.transformed(by: CGAffineTransform(scaleX: scaleX, y: scaleY))
    let context = CIContext()
    guard let cg = context.createCGImage(transformed, from: transformed.extent) else { return nil }
    return UIImage(cgImage: cg)
  }
}
```

```swift
// ConsentViewModel.swift
// 用 Combine 做防抖提交，1.5s 内多次变更仅发送最后一次
import Combine

@MainActor
final class ConsentViewModel: ObservableObject {
  @Published var marketingEmail: Bool = false
  @Published var marketingPush: Bool = false
  @Published var analyticsCookies: Bool = false
  @Published var advertisingCookies: Bool = false
  @Published var lastUpdatedAt: String?

  private let api: APIClient
  private var bag = Set<AnyCancellable>()
  private let subject = PassthroughSubject<Void, Never>()

  init(api: APIClient = .shared) {
    self.api = api
    // 防抖：所有开关变更统一发送到 subject，1.5s 内合并为 1 次请求
    subject
      .debounce(for: .milliseconds(1500), scheduler: DispatchQueue.main)
      .sink { [weak self] in Task { await self?.submit() } }
      .store(in: &bag)
  }

  func bootstrap() async {
    do {
      let r = try await api.getConsent()
      marketingEmail = r.marketingEmail
      marketingPush = r.marketingPush
      analyticsCookies = r.analyticsCookies
      advertisingCookies = r.advertisingCookies
      lastUpdatedAt = r.lastUpdatedAt
    } catch {
      // 静默失败；用户也可手动下拉刷新
    }
  }

  // KeyPath 形式更新 -> 仅做局部变更，再触发防抖
  func update(_ keyPath: WritableKeyPath<ConsentViewModel, Bool>, to value: Bool) {
    self[keyPath: keyPath] = value
    subject.send(())
  }

  private func submit() async {
    let patch = ConsentPatch(
      marketingEmail: marketingEmail,
      marketingPush: marketingPush,
      analyticsCookies: analyticsCookies,
      advertisingCookies: advertisingCookies
    )
    do {
      let r = try await api.putConsent(patch)
      lastUpdatedAt = r.lastUpdatedAt
    } catch {
      // 失败时回滚上一个值（可选）
    }
  }
}

struct ConsentPatch: Codable {
  let marketingEmail: Bool
  let marketingPush: Bool
  let analyticsCookies: Bool
  let advertisingCookies: Bool
}
```

```swift
    var int: UInt64 = 0
    Scanner(string: hex).scanHexInt64(&int)
    let r, g, b: UInt64
    switch hex.count {
    case 6: (r, g, b) = ((int >> 16) & 0xff, (int >> 8) & 0xff, int & 0xff)
    case 8: (r, g, b) = ((int >> 16) & 0xff, (int >> 8) & 0xff, int & 0xff)
    default: (r, g, b) = (255, 255, 255)
    }
    self.init(.sRGB,
              red:   Double(r) / 255,
              green: Double(g) / 255,
              blue:  Double(b) / 255,
              opacity: 1.0)
```
